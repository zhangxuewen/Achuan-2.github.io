Arduino智能小车

发表于2021-04-17|更新于2021-05-27|本科课程
|阅读量:

本次智能小车包括以下三个模块：自动寻迹，超声波测距，蓝牙控制。各模块设计意图如下详述：

  1. 自动寻迹：
    （1）设计方式：在车身下靠近地面处装两个灰度传感器。
    （2） 设计功能：实现小车从出发区沿着白色路面的黑色轨迹到达停车区，要求:
    a. 若两个传感器都没有检测到黑线时，小车直线前进。
    b. 若有一侧传感器检测到黑线时，小车向该侧转弯。
    （3） 设计实现 : 对于不同的情况，在代码中控制不同车轮的转动与停止以及车速。在实际情况中应根据电池电量，车轮损坏，传感器灵敏度等实际情况调整左右轮的转速。
  2. 超声波测距
    （1）设计方式：在车身前上方安装超声波传感器和蜂鸣器
    （2）设计功能：当启动该功能之后小车一边超声测距一边开始向前自动循迹同时响起蜂鸣器，大约离障碍物10cm时即终点线前方，蜂鸣器关闭，小车停止
    （3）设计实现：导入超声波测距模块，通过实际测量障碍物在小车10cm前方时message值，在测距大于10cm时，使用自动循迹函数循迹并响起蜂鸣器，测距小于10cm时，关闭蜂鸣器停止小车，
  3. 蓝牙控制：
    (1) 设计方式：小车内部安装有蓝牙设备，手机安装蓝牙串口通信助手。
    (2) 设计功能：手机通过“蓝牙串口通信助手”软件给智能小车发送相关的遥控指令，控制小车按规定路径到达终点。
    如 f：前进 b：后退 l：左转 r：右转 s：停止
    (3) 设计实现：在代码中为不同指令编写函数，通过控制不同车轮的转动与静止以及转速实现。


代码

int L_state = 0; //定义状态，初始为蓝牙控制模式

void setup() // put your setup code here, to run once:
{
  //初始化串口
  Serial.begin(9600);
  {
    //定义接口类型oid loop()
    pinMode(16, INPUT);  // 左侧灰度传感器端口-16
    pinMode(17, INPUT);  // 右侧灰度传感器端口-17
    pinMode(9, OUTPUT);  // 右电机反转端口-9
    pinMode(10, OUTPUT); // 右电机正转端口-10
    pinMode(5, OUTPUT);  // 左电机反转端口-5
    pinMode(6, OUTPUT);  //左电机正转端口-6
    pinMode(11, OUTPUT); //蜂鸣器端口-11
    //输出清零
    analogWrite(5, 0);
    analogWrite(6, 0);
    analogWrite(9, 0);
    analogWrite(10, 0);
    digitalWrite(15, LOW); // 测距发射端口15,低电平
  }
}
//主函数
void loop()
{
  char getstr = Serial.read(); //读取HC-05蓝牙模块接收的字符
  switch (getstr)
  {
  case 'l': //左转
    drive_left();
    break;
  case 'r': //右转
    drive_right();
    break;
  case 'f': //前进
    drive_forward();
    break;
  case 'b': //后退
    drive_back();
    break;
  case 'd': //减速
    speed_down();
    break;
  case 's': //暂停
    stop();
    break;
  case 't': //自动寻迹
    _trace();
    break;
  case 'i': //超声检测
    _supersound();
    break;
  default:
    //unknow command!
    break;
  }
}
//自动寻迹函数
void _trace()
{
  if ((!(digitalRead(16)) && digitalRead(17))) //如果右侧传感器探测到黑线，向右转
  {
    turn_right();
  }
  else
  {
    if ((!(digitalRead(17)) && digitalRead(16))) //如果左侧传感器探测到黑线，向左转
    {
      turn_left();
    }
    else //两侧传感器均探测到黑线或白线，直行
    {
      drive_forward();
    }
  }
}
//超声波测距函数
int ardublockUltrasonicSensorCodeAutoGeneratedReturnCM(int trigPin, int echoPin)
{
  long duration;
  pinMode(trigPin, OUTPUT);
  pinMode(echoPin, INPUT);
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(20);
  digitalWrite(trigPin, LOW);
  duration = pulseIn(echoPin, HIGH);
  duration = duration / 58;
  if ((duration < 2) || (duration > 300))
    return false;
  return duration; //返回距离
}
//超声检测
void _supersound()
{
  Serial.print("message");
  Serial.print(ardublockUltrasonicSensorCodeAutoGeneratedReturnCM(15, 14));
  Serial.println();
  delay(100);
  //开启超声检测
  if (((ardublockUltrasonicSensorCodeAutoGeneratedReturnCM(15, 14)) <= (45)))
  { // 检测前方有障碍物,message <= 45
    stop();
    noTone(11); //蜂鸣器无声
  }
  else
  { // 前方暂无障碍物，自动循迹前行
    _trace();
    tone(11, 440, 1000); //蜂鸣器启动
  }
}

//小车左转
void drive_left()
{
  analogWrite(5,&nbsp;0);
  analogWrite(6,&nbsp;0);
  analogWrite(9,&nbsp;0);
  analogWrite(10,&nbsp;220); //右边轮子正转
  &nbsp;
}
//小车右转
void drive_right()
{
  analogWrite(5,&nbsp;0);
  analogWrite(6,&nbsp;220); //左边轮子正转
  analogWrite(9,&nbsp;0);
  analogWrite(10,&nbsp;0);
}
// 小车前进
void drive_forward()
{
  analogWrite(5,&nbsp;0);
  analogWrite(6,&nbsp;250);
  analogWrite(9,&nbsp;0);
  analogWrite(10,&nbsp;200);
}
// 小车后退
void drive_back()
{ //左右轮反转
  analogWrite(5, 250);
  analogWrite(6, 0);
  analogWrite(9, 200);
  analogWrite(10, 0);
}
// 小车减速
void speed_down()
{
  analogWrite(5, 0);
  analogWrite(6, 160);
  analogWrite(9, 0);
  analogWrite(10, 160);
}
// 小车停止
void stop()
{
  &nbsp;&nbsp;&nbsp;&nbsp;analogWrite(5,&nbsp;0);
  &nbsp;&nbsp;&nbsp;&nbsp;analogWrite(6,&nbsp;0);
  &nbsp;&nbsp;&nbsp;&nbsp;analogWrite(9,&nbsp;0);
  &nbsp;&nbsp;&nbsp;&nbsp;analogWrite(10,&nbsp;0);
}
